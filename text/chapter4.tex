数式
\begin{align}
    \begin{array}{rrr}
      \displaystyle	
        Q_{ij} = \frac{D_{ij}}{L_{ij}}(P_i-P_j)
    \end{array} \label{array:stream}
\end{align}
\begin{align}
    \begin{array}{rrr}
      \displaystyle	
      % fのバリエーションについて注意する
        \frac{d}{dt}D_{ij}=f(|Q_{ij}|) -aD_{ij}
    \end{array} \label{array:tube}
\end{align}
\begin{align}
  \begin{array}{rrr}
    \displaystyle	
      f(|Q_{ij}|) = |Q_{ij}|^\gamma
  \end{array} \label{array:gamma}
\end{align}
